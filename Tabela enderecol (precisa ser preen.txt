‚úÖ 2. Consultas com JOIN (INNER, LEFT, RIGHT)
üîç 1. INNER JOIN entre pedido e produto
SELECT pedido.pedido_id, produto.titulo, produto.preco
FROM pedido
INNER JOIN produto ON pedido.produto_id = produto.produto_id;

üîç 2. LEFT JOIN entre cadastroc e enderecol
SELECT c.nome, c.sobrenome, e.rua, e.cidade
FROM cadastroc c
LEFT JOIN enderecol e ON c.endereco_id = e.endereco_id;

üîç 3. RIGHT JOIN entre conta e pedido
SELECT c.conta_id, p.pedido_id, p.produto_id
FROM conta c
RIGHT JOIN pedido p ON c.pedido_id = p.pedido_id;

üîç 4. INNER JOIN entre pagamentop, pedido e produto
SELECT pg.pagamento_id, pg.forma_pagamento, pr.titulo
FROM pagamentop pg
INNER JOIN pedido p ON pg.pedido_id = p.pedido_id
INNER JOIN produto pr ON p.produto_id = pr.produto_id
